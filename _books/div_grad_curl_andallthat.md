DIVERGENCE THEOREM

\begin{equation}
\iint_S F\cdot\hat{n}dS=\iiint_V\nabla\cdot F dV
\end{equation}

STOKES' THEOREM

\begin{equation}
\oint_C F \cdot \hat{t} ds = \iint_S \hat{n} \cdot \nabla \times F ds
\end{equation}
